1.添加Swagger依赖

  <dependency>
            <groupId>org.springdoc</groupId>
            <artifactId>springdoc-openapi-ui</artifactId>
            <version>1.3.9</version>
            <scope>compile</scope>
   </dependency>

2. 新建SwaggerConfig配置类

@Configuration
@Profile({"local", "dev", "sit"})
public class SwaggerConfig {
    @Bean
    public GroupedOpenApi DemandBaseApi(){
        return GroupedOpenApi.builder()
                .setGroup("分布式ID获取接口")
                .pathsToMatch("/id/segment/**")
                .packagesToScan("com.shelton.user.account.idcenter.controller")
                .build();
    }

    @Bean
    public OpenAPI openAPI(){
        Contact contact = new Contact();
        contact.setName("Shelton_Lee");
        contact.setEmail("mr.lixiaosheng@gmail.com");
        return new OpenAPI()
                .components(new Components().addSecuritySchemes("bearer-jwt", new SecurityScheme().type(SecurityScheme.Type.APIKEY)
                        .in(SecurityScheme.In.HEADER).name("Authorization")))
                .addSecurityItem(new SecurityRequirement().addList("bearer-jwt", Arrays.asList("read", "write")))
                .info(new Info().title("分布式ID获取接口")
                        .version("v1").contact(contact)
                        .description("分布式ID获取"));
    }
}

3. 注解使用

  • 类

    @Api(description = "帖在类上，整个controller的注释", tags = "xxxxxController")
@ApiModel   //标记类时swagger的解析类,比如响应类

/*用于响应类上，表示一个返回响应数据的信息
            （这种一般用在post创建的时候，使用@RequestBody这样的场景，
            请求参数无法使用@ApiImplicitParam注解进行描述的时候）
    @ApiModelProperty：用在属性上，描述响应类的属性 */

  • 方法

    @ApiOperation(value = "贴在方法上，方法的注释")

@ApiImplicitParams   //用在方法上表明一组参数的说明
@ApiImplicitParam  //配合上述注解，表明单个参数

//针对上述注解的一个参数说明
paramType：参数放在哪个地方
    header-->请求参数的获取：@RequestHeader  String idstr = request.getHeader("id");
    query-->请求参数的获取：@RequestParam    @RequestParam(defaultValue = "STC") String tenantCode
    path（用于restful接口）-->请求参数的获取：@PathVariable  @PathVariable(name = "id") Long id
    body（不常用）
    form（不常用）

@ApiResponses //表示一组响应
@ApiResponse

//示例
